# 深考buffer

const allocUnsafe = Buffer?Buffer.allocUnsafe:undefined;//判断是否有buffer
function cloneBuffer(buffer,isDeep){
    if(!isDeep){
        return buffer.slice()
    }
    const length = buffer.length()
    const result = allocUnsafe?allocUnsafe(length):new buffer.constructor(length);
    buffer.copy(result)
    return result
}
